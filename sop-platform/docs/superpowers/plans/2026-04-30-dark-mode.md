## Class Replacement Reference

> **`text-secondary` name:** This project uses no custom Tailwind color palette, so `text-secondary` does not conflict with any built-in or plugin class. Safe to use.

| Replace this | With this | Notes |
|---|---|---|
| `bg-gray-50` (page bg) | `bg-page` | Outermost page/layout backgrounds |
| `bg-white` (card/panel) | `bg-card` | Cards, modals, header, sidebars |
| `bg-gray-100` | `bg-raised` | Badge backgrounds, hover surfaces |
| `bg-white` on `<input>` / `<textarea>` / form field wrappers | `bg-input` | Input fields specifically |
| `border-gray-200` | `border-default` | Standard dividers and borders |
| `border-gray-100` | `border-subtle` | Very faint borders |
| `text-gray-900` | `text-default` | Headings, strong labels |
| `text-gray-800` | `text-secondary` | Body text |
| `text-gray-700` | `text-secondary` | Body text |
| `text-gray-600` | `text-muted` | Metadata, helper text |
| `text-gray-500` | `text-muted` | Muted labels |
| `hover:bg-gray-50` | `hover:bg-raised` | Hover state surface elevation |
| `hover:bg-gray-100` | `hover:bg-raised` | Hover state surface elevation |

**Do NOT replace:**
- `text-gray-400` and dimmer — intentionally faint; keep as-is
- Status badge colors (`bg-green-*`, `bg-amber-*`, `bg-red-*`, etc.)
- Accent classes (`text-violet-*`, `text-blue-*`, `bg-gradient-*`)

## Task 4 — Create `ThemeContext.tsx`

**File:** `frontend/src/contexts/ThemeContext.tsx` (new file)

```tsx
import { createContext, useContext, useEffect, useState, type ReactNode } from 'react'

export type Theme = 'light' | 'dark' | 'gray'

const STORAGE_KEY = 'sop-theme'

interface ThemeContextValue {
  theme: Theme
  setTheme: (t: Theme) => void
}

const ThemeContext = createContext<ThemeContextValue | null>(null)

export function ThemeProvider({ children }: { children: ReactNode }) {
  const [theme, setThemeState] = useState<Theme>(
    () => (localStorage.getItem(STORAGE_KEY) as Theme | null) ?? 'light',
  )

  useEffect(() => {
    document.documentElement.dataset.theme = theme
    localStorage.setItem(STORAGE_KEY, theme)
  }, [theme])

  return (
    <ThemeContext.Provider value={{ theme, setTheme: setThemeState }}>
      {children}
    </ThemeContext.Provider>
  )
}

export function useTheme(): ThemeContextValue {
  const ctx = useContext(ThemeContext)
  if (!ctx) throw new Error('useTheme must be used within ThemeProvider')
  return ctx
}
```

## Task 7 — Batch replace in all component + route files

Run this PowerShell command from `frontend/src/` to do the mechanical replacements across all files:

```powershell
$dir = "d:/CloudNavision/1. Projects/SOP/SOP Automation System/sop-platform/frontend/src"

$replacements = @(
  @{ From = ' bg-gray-50';  To = ' bg-page'   },
  @{ From = '"bg-gray-50"'; To = '"bg-page"'  },
  @{ From = ' bg-white';    To = ' bg-card'   },
  @{ From = '"bg-white"';   To = '"bg-card"'  },
  @{ From = ' bg-gray-100'; To = ' bg-raised' },
  @{ From = 'border-gray-200'; To = 'border-default' },
  @{ From = 'border-gray-100'; To = 'border-subtle'  },
  @{ From = ' text-gray-900'; To = ' text-default'   },
  @{ From = ' text-gray-800'; To = ' text-secondary' },
  @{ From = ' text-gray-700'; To = ' text-secondary' },
  @{ From = ' text-gray-600'; To = ' text-muted'     },
  @{ From = ' text-gray-500'; To = ' text-muted'     },
  @{ From = 'hover:bg-gray-50';  To = 'hover:bg-raised' },
  @{ From = 'hover:bg-gray-100'; To = 'hover:bg-raised' }
)

Get-ChildItem -Path $dir -Recurse -Include "*.tsx","*.ts" | ForEach-Object {
  $content = Get-Content $_.FullName -Raw
  $original = $content
  foreach ($r in $replacements) {
    $content = $content.Replace($r.From, $r.To)
  }
  if ($content -ne $original) {
    Set-Content $_.FullName $content -NoNewline -Encoding utf8
    Write-Host "Updated: $($_.Name)"
  }
}
```
